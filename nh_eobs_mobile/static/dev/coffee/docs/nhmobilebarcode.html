        event.preventDefault()
        setTimeout( ->
          self.barcode_scanned(self, event)
        , 1000)

    self.input.addEventListener 'keypress', (event) ->
      if event.keyCode is 13 or event.keyCode is 0 or event.keyCode is 116
        event.preventDefault()
        self.barcode_scanned(self, event)
    self.input.focus()
